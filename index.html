Запасные курсы валют (если нет интернета) let exchangeRates = { "USD": 1.09, "GBP": 0.86, "UAH": 44.10, "JPY": 170.50 }; // Функция для автообновления курсов через интернет (Вариант 3) async function updateRates() { try { // Запрашиваем свежие курсы относительно EUR const response = await fetch('https://open.er-api.com/v6/latest/EUR'); if (!response.ok) throw new Error('Ошибка сети'); const data = await response.json(); // Обновляем наши переменные свежими данными из интернета exchangeRates.USD = data.rates.USD || 1.09; exchangeRates.GBP = data.rates.GBP || 0.86; exchangeRates.UAH = data.rates.UAH || 44.10; exchangeRates.JPY = data.rates.JPY || 170.50; console.log("Курсы валют успешно обновлены:", exchangeRates); } catch (error) { console.log("Не удалось загрузить курсы, используем стандартные:", error); } } // Запускаем обновление курсов сразу при загрузке страницы updateRates(); // 2. Функция звукового отклика (Вариант 4) function playClickSound() { try { const audioCtx = new (window.AudioContext || window.webkitAudioContext)(); const oscillator = audioCtx.createOscillator(); const gainNode = audioCtx.createGain(); oscillator.type = 'sine'; // Мягкий футуристичный звук oscillator.frequency.setValueAtTime(1200, audioCtx.currentTime); // Высокий тон (клик) // Быстрое затухание звука, чтобы получился именно клик gainNode.gain.setValueAtTime(0.05, audioCtx.currentTime); gainNode.gain.exponentialRampToValueAtTime(0.00001, audioCtx.currentTime + 0.05); oscillator.connect(gainNode); gainNode.connect(audioCtx.destination); oscillator.start(); oscillator.stop(audioCtx.currentTime + 0.05); } catch (e) { console.log("Звук не поддерживается браузером"); } } // Переменные для логики калькулятора let currentInput = ""; const display = document.querySelector('.display'); // 3. Функция вывода на экран с анимацией (Вариант 1) function updateDisplay(text) { // Добавляем класс анимации (быстрое мигание) display.style.transition = "opacity 0.05s ease-in-out"; display.style.opacity = "0.3"; setTimeout(() => { display.innerText = text; display.style.opacity = "1"; }, 50); } // Главная функция нажатия на кнопки function pressKey(key) { // Включаем звук при каждом тапе! playClickSound(); if (key === 'C') { currentInput = ""; updateDisplay("0"); return; } if (key === '.') { if (!currentInput.includes('.')) { currentInput += '.'; updateDisplay(currentInput); } return; } // Логика конвертации валют if (key === 'USD' || key === 'EUR' || key === 'GBP' || key === 'UAH' || key === 'JPY') { let amount = parseFloat(currentInput); if (isNaN(amount) || amount <= 0) { updateDisplay("Введите сумму"); return; } // Если нажали евро, то считаем, что ввели евро и переводим в другие валюты if (key === 'EUR') { let usd = (amount * exchangeRates.USD).toFixed(2); let gbp = (amount * exchangeRates.GBP).toFixed(2); let uah = (amount * exchangeRates.UAH).toFixed(2); let jpy = (amount * exchangeRates.JPY).toFixed(2); updateDisplay(`${amount} € равно:\n ${usd} $ | ${gbp} £ | ${uah} грн | ${jpy} ¥`); } else { // Если выбрали другую валюту, можно дописать логику обратного перевода, // но пока оставляем твою проверенную схему! updateDisplay(`Нажми € для перевода`); } currentInput = ""; // Сбрасываем ввод после расчета return; } // Ввод цифр if (currentInput === "0") currentInput = ""; currentInput += key; updateDisplay(currentInput); } // Привязываем клики к кнопкам (убедись, что на кнопках в HTML стоит onclick="pressKey('...')")
